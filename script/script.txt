// SCRIPT SEM INFLUÊNCIA DO DIA DE HJ
// Voltar um página
function goBack() {
    window.history.back();
}
// Seleciona todos os botões de tag e todos os itens de eventos
const tagButtons = document.querySelectorAll(".tag-btn");
const eventItems = document.querySelectorAll(".itens");

// Função para filtrar eventos por tag
function filterEvents(selectedTag) {
    // Oculta todos os itens de evento
    eventItems.forEach(item => {
        item.classList.add("hide");
    });

    // Mostra apenas os itens de evento do dia selecionado
    const selectedItems = document.querySelectorAll(`[data-tag="${selectedTag}"]`);
    selectedItems.forEach(item => {
        item.classList.remove("hide");
    });

    // Remove a classe "selected" de todos os botões e adiciona ao botão clicado
    tagButtons.forEach(btn => {
        btn.classList.remove("selected");
    });
}

// Filtra eventos quando a página é carregada
document.addEventListener("DOMContentLoaded", function () {
    const initialTag = "seg"; // Dia inicial para exibição
    filterEvents(initialTag); // Filtra eventos inicialmente por "seg"

    // Encontra o botão correspondente ao dia inicial e adiciona a classe "selected"
    const initialButton = document.querySelector(`.tag-btn[data-tag="${initialTag}"]`);
    initialButton.classList.add("selected");
});

// Adiciona um evento de clique a todos os botões de tag
tagButtons.forEach(button => {
    button.addEventListener("click", function () {
        const selectedTag = this.getAttribute("data-tag");
        filterEvents(selectedTag);
        this.classList.add("selected");
    });
});
